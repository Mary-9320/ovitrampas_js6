        const { useState, useEffect, useMemo, useRef } = React;

        // --- COMPONENTE DEL MAPA REAL (LEAFLET) ---
        function MapaInteractivo({ data, onMarkerClick }) {
            const mapRef = useRef(null);
            const mapInstance = useRef(null);

            useEffect(() => {
                if (!mapInstance.current && mapRef.current) {
                    mapInstance.current = L.map(mapRef.current, {
                        zoomControl: true,
                        attributionControl: false 
                    }).setView([25.3, -100.0], 12);

                    L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
                        subdomains: 'abcd',
                        maxZoom: 19,
                        crossOrigin: true 
                    }).addTo(mapInstance.current);
                }

                if (mapInstance.current) {
                    mapInstance.current.eachLayer((layer) => {
                        if (layer instanceof L.CircleMarker) {
                            mapInstance.current.removeLayer(layer);
                        }
                    });
                }

                if (data && data.length > 0 && mapInstance.current) {
                    const bounds = L.latLngBounds();
                    
                    data.forEach(punto => {
                        if (punto.y && punto.x) {
                            const color = punto.sectorRiesgo === 'Crítico' ? '#f43f5e' : 
                                          punto.sectorRiesgo === 'Alto' ? '#fb923c' :    
                                          '#0891b2';                                     

                            const marker = L.circleMarker([punto.y, punto.x], {
                                radius: 6,
                                fillColor: color,
                                color: color,
                                weight: 1,
                                opacity: 1,
                                fillOpacity: 0.8
                            }).addTo(mapInstance.current);

                            const popupContent = `
                                <div style="font-family: sans-serif; min-width: 150px;">
                                    <p style="font-weight: bold; color: #22d3ee; margin: 0 0 4px 0; border-bottom: 1px solid #334155; padding-bottom: 4px;">${punto.id}</p>
                                    <p style="font-size: 11px; margin: 0 0 2px 0;">S: ${punto.sector} | M: ${punto.manzana}</p>
                                    <p style="font-size: 11px; margin: 0 0 4px 0; color: ${color}; font-weight: bold;">Riesgo Sector: ${punto.sectorRiesgo}</p>
                                    <p style="font-size: 12px; margin: 0;">Huevos Trampa: <b>${punto.huevos}</b></p>
                                </div>
                            `;
                            marker.bindPopup(popupContent);
                            marker.on('click', () => { if(onMarkerClick) onMarkerClick(punto); });
                            bounds.extend([punto.y, punto.x]);
                        }
                    });
                    
                    if (bounds.isValid()) {
                        mapInstance.current.fitBounds(bounds, { padding: [30, 30] });
                    }
                }
            }, [data]);

            return <div ref={mapRef} style={{ width: '100%', height: '100%', borderRadius: '0.75rem', zIndex: 1 }}></div>;
        }

        // --- SISTEMA PRINCIPAL ---
        function CentroInteligenciaVectorial() {
            const [activeTab, setActiveTab] = useState('upload'); 
            
            const [geoCatalog, setGeoCatalog] = useState({}); 
            const [isGeoLoaded, setIsGeoLoaded] = useState(false);
            const [isProcessing, setIsProcessing] = useState(false);
            
            // HISTÓRICO DE SEMANAS
            const [historialSemanas, setHistorialSemanas] = useState({});
            const [semanaActual, setSemanaActual] = useState(null);

            const [selectedHotspot, setSelectedHotspot] = useState(null);
            const [mapViewMode, setMapViewMode] = useState('puntos'); 
            const [trendSector, setTrendSector] = useState('');

            const mapContainerRef = useRef(null);

            // VARIABLES DE LA SEMANA ACTUAL
            const dataSemanaActual = semanaActual ? historialSemanas[semanaActual] : null;
            const gridData = dataSemanaActual?.gridData || [];
            
            // STATS CON NOMENCLATURA OFICIAL CENAPRECE
            const statsGlobales = dataSemanaActual?.statsGlobales || {
                ovitrampasRevisadas: 0, ovitrampasPositivas: 0, totalHuevos: 0, manzanasRevisadas: 0, manzanasPositivas: 0, semanaEpidemiologica: 'S/N'
            };
            const datosPorSector = dataSemanaActual?.datosPorSector || [];
            const isWeeklyLoaded = Object.keys(historialSemanas).length > 0;
            
            // LECTURA DE PADRÓN GEOGRÁFICO
            const handleGeoUpload = (e) => {
                const file = e.target.files[0];
                if (!file) return;
                const reader = new FileReader();
                reader.onload = (evt) => {
                    const rows = evt.target.result.split('\n');
                    const headers = rows[0].split(',').map(h => h.trim().toLowerCase());
                    const xIdx = headers.findIndex(h => h === 'pocision_x' || h === 'posicion_x');
                    const yIdx = headers.findIndex(h => h === 'pocision_y' || h === 'posicion_y');
                    const oviIdx = headers.findIndex(h => h === 'ovitrampa');
                    const catalog = {};
                    for(let i = 1; i < rows.length; i++) {
                        const cols = rows[i].split(',');
                        if(cols.length > Math.max(xIdx, yIdx, oviIdx)) {
                            const id = cols[oviIdx]?.trim();
                            const x = parseFloat(cols[xIdx]);
                            const y = parseFloat(cols[yIdx]);
                            if(id && !isNaN(x) && !isNaN(y)) catalog[id] = { x, y };
                        }
                    }
                    setGeoCatalog(catalog);
                    setIsGeoLoaded(true);
                };
                reader.readAsText(file);
            };

            // LECTURA DE DATOS MENSUALES/SEMANALES
            const handleWeeklyUpload = (e) => {
                const file = e.target.files[0];
                if (!file) return;
                setIsProcessing(true);

                const reader = new FileReader();
                reader.onload = (evt) => {
                    const rows = evt.target.result.split('\n');
                    if(rows.length <= 1) { setIsProcessing(false); return; }
                    const headers = rows[0].split(',').map(h => h.trim().toLowerCase());
                    
                    const oviIdx = headers.findIndex(h => h === 'ovitrampa');
                    const huevosIdx = headers.findIndex(h => h === 'huevecillos');
                    const sectorIdx = headers.findIndex(h => h === 'sector');
                    const mznIdx = headers.findIndex(h => h === 'manzana');
                    const muniIdx = headers.findIndex(h => h === 'municipio');
                    const semIdx = headers.findIndex(h => h === 'semana epidemiologica' || h === 'semana');

                    const weeksData = {};

                    for(let i = 1; i < rows.length; i++) {
                        const cols = rows[i].split(',');
                        if(cols.length > Math.max(oviIdx, huevosIdx)) {
                            const id = cols[oviIdx]?.trim();
                            const huevosStr = cols[huevosIdx]?.trim();
                            if(!id || huevosStr === undefined || huevosStr === '') continue;
                            const huevos = parseInt(huevosStr, 10);
                            if(isNaN(huevos)) continue;
                            
                            const semana = (semIdx !== -1 && cols[semIdx]) ? cols[semIdx].trim() : 'S/N';

                            if (!weeksData[semana]) {
                                weeksData[semana] = {
                                    mergedData: [], totalHuevos: 0, positivas: 0, revisadas: 0, manzanasTracker: {}, sectoresTracker: {}
                                };
                            }

                            const weekObj = weeksData[semana];
                            
                            // Nomenclatura oficial: Ovitrampas Revisadas
                            weekObj.revisadas++;
                            if (huevos > 0) weekObj.positivas++;
                            weekObj.totalHuevos += huevos;

                            const sector = cols[sectorIdx]?.trim() || 'S/N';
                            const manzana = cols[mznIdx]?.trim() || 'S/N';
                            const muni = cols[muniIdx]?.trim() || 'Sector';
                            
                            const mznKey = `${sector}-${manzana}`;
                            if (!weekObj.manzanasTracker[mznKey]) weekObj.manzanasTracker[mznKey] = { huevos: 0, positivas: 0 };
                            weekObj.manzanasTracker[mznKey].huevos += huevos;
                            if (huevos > 0) weekObj.manzanasTracker[mznKey].positivas++;

                            const secKey = `${muni} - Sector ${sector}`;
                            if (!weekObj.sectoresTracker[secKey]) weekObj.sectoresTracker[secKey] = { nombre: secKey, municipio: muni, sector: sector, huevos: 0, positivas: 0, revisadas: 0, manzanasSet: new Set() };
                            weekObj.sectoresTracker[secKey].revisadas++;
                            weekObj.sectoresTracker[secKey].huevos += huevos;
                            if (huevos > 0) weekObj.sectoresTracker[secKey].positivas++;
                            weekObj.sectoresTracker[secKey].manzanasSet.add(manzana);

                            const geo = geoCatalog[id];
                            if (geo) {
                                weekObj.mergedData.push({ id, municipio: muni, sector, manzana, x: geo.x, y: geo.y, huevos });
                            }
                        }
                    }

                    const newHistorial = { ...historialSemanas };
                    let maxSemana = semanaActual || 0;

                    Object.keys(weeksData).forEach(semana => {
                        const weekObj = weeksData[semana];
                        // Nomenclatura oficial: Manzanas Revisadas
                        const manzanasRevisadas = Object.keys(weekObj.manzanasTracker).length;
                        const manzanasPositivas = Object.values(weekObj.manzanasTracker).filter(m => m.positivas > 0).length;

                        const datosSectorArray = Object.values(weekObj.sectoresTracker).map(s => {
                            const pho = s.positivas > 0 ? (s.huevos / s.positivas).toFixed(1) : 0;
                            const phm = s.manzanasSet.size > 0 ? (s.huevos / s.manzanasSet.size).toFixed(1) : 0; 
                            const riesgo = parseFloat(phm) > 50 ? 'Crítico' : parseFloat(phm) > 20 ? 'Alto' : 'Bajo';
                            return { ...s, pho: parseFloat(pho), phm: parseFloat(phm), riesgoNivel: riesgo, manzanasUnicas: s.manzanasSet.size };
                        }).sort((a, b) => b.phm - a.phm);

                        const calcPHMGlobal = manzanasRevisadas > 0 ? (weekObj.totalHuevos / manzanasRevisadas).toFixed(1) : 0;

                        newHistorial[semana] = {
                            gridData: weekObj.mergedData,
                            statsGlobales: {
                                ovitrampasRevisadas: weekObj.revisadas,
                                ovitrampasPositivas: weekObj.positivas,
                                totalHuevos: weekObj.totalHuevos,
                                manzanasRevisadas: manzanasRevisadas,
                                manzanasPositivas: manzanasPositivas,
                                semanaEpidemiologica: semana,
                                phmGlobal: parseFloat(calcPHMGlobal)
                            },
                            datosPorSector: datosSectorArray
                        };

                        if (semana !== 'S/N' && parseInt(semana) > parseInt(maxSemana || 0)) {
                            maxSemana = semana;
                        }
                    });

                    setTimeout(() => {
                        setHistorialSemanas(newHistorial);
                        if (maxSemana) setSemanaActual(maxSemana);
                        setIsProcessing(false);
                        if (activeTab === 'upload') setActiveTab('tactica'); 
                    }, 1000);
                };
                reader.readAsText(file);
                e.target.value = null; 
            };

            // CÁLCULOS INDICADORES GLOBALES DE LA SEMANA ACTUAL (Acorde a Manual CENAPRECE)
            const calcIPO = statsGlobales.ovitrampasRevisadas > 0 ? ((statsGlobales.ovitrampasPositivas / statsGlobales.ovitrampasRevisadas) * 100).toFixed(1) : '0.0';
            const calcPHO = statsGlobales.ovitrampasPositivas > 0 ? (statsGlobales.totalHuevos / statsGlobales.ovitrampasPositivas).toFixed(1) : '0.0';
            const calcPHM = statsGlobales.manzanasRevisadas > 0 ? (statsGlobales.totalHuevos / statsGlobales.manzanasRevisadas).toFixed(1) : '0.0';
            const calcPHMP = statsGlobales.manzanasPositivas > 0 ? (statsGlobales.totalHuevos / statsGlobales.manzanasPositivas).toFixed(1) : '0.0';
            const calcMP = statsGlobales.manzanasRevisadas > 0 ? ((statsGlobales.manzanasPositivas / statsGlobales.manzanasRevisadas) * 100).toFixed(1) : '0.0';

            const mapDataWithSectorRisk = useMemo(() => {
                return gridData.map(punto => {
                    const sectorName = `${punto.municipio} - Sector ${punto.sector}`;
                    const sectorInfo = datosPorSector.find(s => s.nombre === sectorName);
                    return { ...punto, sectorPhm: sectorInfo ? sectorInfo.phm : 0, sectorRiesgo: sectorInfo ? sectorInfo.riesgoNivel : 'Bajo' };
                });
            }, [gridData, datosPorSector]);

            const generarAnalisisDescriptivo = () => {
                const ipo = parseFloat(calcIPO); const pho = parseFloat(calcPHO); const phm = parseFloat(calcPHM); const mp = parseFloat(calcMP);

                let txtTerritorial = "";
                if (mp > 20 && ipo > 15) txtTerritorial = "Ambos porcentajes (%MP e IPO) son altos. Esto confirma que la infestación no es un problema aislado en un solo rincón, sino que el vector ya se ha dispersado por múltiples cuadras y ha logrado invadir una gran cantidad de viviendas en la zona evaluada.";
                else if (mp > 20) txtTerritorial = "El Porcentaje de Manzanas Positivas (%MP) indica una dispersión amplia a lo largo de las cuadras, aunque la invasión al interior de las viviendas (IPO) se mantiene moderada o focalizada.";
                else if (ipo > 15) txtTerritorial = "Aunque el vector no ha abarcado un número extenso de manzanas, las viviendas dentro de los polígonos afectados tienen una alta tasa de positividad (IPO).";
                else txtTerritorial = "Tanto el IPO como el %MP se encuentran en niveles controlados. La infestación se mantiene en áreas focalizadas sin una dispersión habitacional generalizada.";

                let txtDensidad = "";
                if (pho > 40 && ipo > 15) txtDensidad = "Se detecta un PHO alto concurrente con un IPO elevado. Esto significa que la plaga no solo está extendida geográficamente, sino que en cada punto infestado existe una concentración masiva de mosquitos hembra reproduciéndose.";
                else if (pho > 40) txtDensidad = "A pesar de una menor dispersión territorial, los focos que dieron positivo registran un PHO alarmante, indicando una concentración masiva de oviposición en sitios específicos (alta productividad focal).";
                else txtDensidad = "El Promedio de Huevos por Ovitrampa (PHO) muestra una tasa de oviposición de leve a moderada en los puntos positivos.";

                let txtRiesgo = ""; let riesgoNivelClase = "text-cyan-400"; let riesgoNivelTexto = "BAJO / CONTROLADO";
                if (phm >= 50) {
                    txtRiesgo = "El PHM se encuentra en niveles de ALERTA MÁXIMA. Se recomienda implementar urgente un Control de Barrido e Intervención Integral. Correlacionar inmediatamente con casos probables de dengue para justificar control químico.";
                    riesgoNivelClase = "text-rose-500 font-black"; riesgoNivelTexto = "CRÍTICO";
                } else if (phm >= 20) {
                    txtRiesgo = "El PHM evidencia un Riesgo de Transmisión ACTIVO. Se requiere priorizar Operativos de Barrido en los sectores marcados en el mapa antes de un brote epidémico.";
                    riesgoNivelClase = "text-orange-400 font-bold"; riesgoNivelTexto = "ALTO";
                } else {
                    txtRiesgo = "El PHM sugiere un escenario de baja probabilidad para el sostenimiento de una cadena de transmisión. Mantener Tratamientos Focalizados en donde se hayan detectado ovitrampas positivas.";
                    riesgoNivelClase = "text-cyan-400 font-bold"; riesgoNivelTexto = "BAJO";
                }
                return { txtTerritorial, txtDensidad, txtRiesgo, riesgoNivelClase, riesgoNivelTexto };
            };
            const analisisTextos = generarAnalisisDescriptivo();

            // PREPARACIÓN DE DATOS PARA PESTAÑA "TENDENCIAS"
            const allSectorsList = useMemo(() => {
                const set = new Set();
                Object.values(historialSemanas).forEach(week => week.datosPorSector.forEach(s => set.add(s.nombre)));
                return Array.from(set).sort();
            }, [historialSemanas]);

            useEffect(() => {
                if(allSectorsList.length > 0 && !trendSector) setTrendSector(allSectorsList[0]);
            }, [allSectorsList]);

            const chartTrendData = useMemo(() => {
                const semanasOrdenadas = Object.keys(historialSemanas).filter(k => k !== 'S/N').sort((a,b) => parseInt(a) - parseInt(b));
                return semanasOrdenadas.map(sem => {
                    const weekData = historialSemanas[sem];
                    const sectorData = weekData.datosPorSector.find(s => s.nombre === trendSector);
                    return {
                        semana: `SE ${sem}`,
                        PHM_Global: weekData.statsGlobales.phmGlobal,
                        PHM_Sector: sectorData ? sectorData.phm : 0
                    };
                });
            }, [historialSemanas, trendSector]);

            const descargarIndicadoresCSV = () => {
                let csvContent = "data:text/csv;charset=utf-8,";
                csvContent += "Semana,Indicador,Valor,Formula\n";
                csvContent += `${statsGlobales.semanaEpidemiologica},% de Ovitrampas Positivas (IPO),${calcIPO}%,(Ovitrampas Positivas / Ovitrampas Revisadas) X 100\n`;
                csvContent += `${statsGlobales.semanaEpidemiologica},Promedio de Huevos por Ovitrampa (PHO),${calcPHO},(Total de Huevos / Ovitrampas Positivas)\n`;
                csvContent += `${statsGlobales.semanaEpidemiologica},Promedio de Huevos por Manzana (PHM),${calcPHM},(Total de Huevos / Manzanas Revisadas)\n`;
                csvContent += `${statsGlobales.semanaEpidemiologica},Promedio de Huevos por Mzn Positiva (PHMP),${calcPHMP},(Total de Huevos / Manzanas Positivas)\n`;
                csvContent += `${statsGlobales.semanaEpidemiologica},% de Manzanas Positivas (%MP),${calcMP}%,(Manzanas Positivas / Manzanas Revisadas) X 100\n\n`;
                csvContent += "Municipio,Sector,Ovitrampas Revisadas,Manzanas Revisadas,Positivas,Total Huevos,PHO Sector,PHM Sector,Nivel Riesgo (Basado en PHM)\n";
                datosPorSector.forEach(s => {
                    csvContent += `${s.municipio},${s.sector},${s.revisadas},${s.manzanasUnicas},${s.positivas},${s.huevos},${s.pho},${s.phm},${s.riesgoNivel}\n`;
                });

                const encodedUri = encodeURI(csvContent);
                const link = document.createElement("a");
                link.setAttribute("href", encodedUri);
                link.setAttribute("download", `Indicadores_Vectores_SE${statsGlobales.semanaEpidemiologica}.csv`);
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
            };

            const descargarMapaImagen = async () => {
                if (mapContainerRef.current) {
                    try {
                        const canvas = await html2canvas(mapContainerRef.current, {
                            useCORS: true, allowTaint: true, backgroundColor: '#020617', scale: 2 
                        });
                        const link = document.createElement("a");
                        link.href = canvas.toDataURL("image/png");
                        link.download = `Mapa_Riesgo_Georreferenciado_SE${statsGlobales.semanaEpidemiologica}.png`;
                        link.click();
                    } catch (err) { alert("Error al capturar el mapa. Intenta de nuevo."); }
                }
            };

            const descargarAccionTactica = (tipoAccion, nivelesRiesgo, filename, soloPositivas) => {
                let puntosAfectados = mapDataWithSectorRisk.filter(p => nivelesRiesgo.includes(p.sectorRiesgo));

                if (soloPositivas) {
                    puntosAfectados = puntosAfectados.filter(p => p.huevos > 0);
                }

                if (puntosAfectados.length === 0) {
                    const motivo = soloPositivas ? 'positivas' : 'para el nivel de riesgo seleccionado';
                    alert(`No se detectaron manzanas ${motivo} en la SE ${statsGlobales.semanaEpidemiologica} para ejecutar esta acción táctica.`);
                    return;
                }

                let csvContent = "data:text/csv;charset=utf-8,";
                csvContent += `ORDEN DE TRABAJO: ${tipoAccion} - SE ${statsGlobales.semanaEpidemiologica}\n\n`;
                csvContent += "Ovitrampa,Municipio,Sector,Manzana,Latitud,Longitud,Huevos,Nivel Riesgo Sector\n";

                puntosAfectados.forEach(p => {
                    csvContent += `${p.id},${p.municipio},${p.sector},${p.manzana},${p.y},${p.x},${p.huevos},${p.sectorRiesgo}\n`;
                });

                const encodedUri = encodeURI(csvContent);
                const link = document.createElement("a");
                link.setAttribute("href", encodedUri);
                link.setAttribute("download", `${filename}_SE${statsGlobales.semanaEpidemiologica}.csv`);
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
            };

            return (
                <div className="h-screen w-full bg-slate-950 text-slate-200 font-sans flex flex-col selection:bg-cyan-500 selection:text-white overflow-hidden">
                
                {/* --- BARRA DE NAVEGACIÓN SUPERIOR --- */}
                <nav className="w-full bg-slate-900 border-b border-slate-800 flex items-center justify-between px-4 py-3 z-50 shrink-0 shadow-lg">
                    <div className="flex items-center gap-3 shrink-0 mr-6">
                        <div className="p-2 bg-cyan-500/10 rounded-lg border border-cyan-500/20">
                            <i className="fa-solid fa-bug text-cyan-400 text-xl"></i>
                        </div>

                        </div>
                        </div>
                    )}

                    {/* VISTA 6: RADAR MAPA REAL */}
                    {activeTab === 'radar' && (
                        <div className="flex flex-col xl:flex-row gap-6 animate-fade-in h-[calc(100vh-220px)] min-h-[600px] pb-8">
                        
                        <div className="flex-1 bg-slate-900/50 p-4 rounded-2xl border border-slate-800 backdrop-blur-sm flex flex-col h-[500px] xl:h-full relative">
                            <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center mb-4 gap-4">
                                <h2 className="text-xl font-bold text-white flex items-center gap-2"><i className="fa-solid fa-earth-americas text-cyan-400"></i> Cartografía Táctica (PHM)</h2>
                                <div className="flex flex-wrap gap-2 w-full sm:w-auto">
                                    {mapViewMode === 'puntos' && (
                                        <button onClick={descargarMapaImagen} className="flex-1 sm:flex-none justify-center px-3 py-1.5 rounded-lg text-sm font-bold transition-colors bg-emerald-600 text-white hover:bg-emerald-500 shadow-md flex items-center gap-2">
                                            <i className="fa-solid fa-camera"></i> Foto
                                        </button>
                                    )}
                                    <button onClick={() => setMapViewMode('puntos')} className={`flex-1 sm:flex-none justify-center px-3 py-1.5 rounded-lg text-sm font-bold transition-colors ${mapViewMode === 'puntos' ? 'bg-cyan-600 text-white' : 'bg-slate-800 text-slate-400 hover:bg-slate-700'}`}>GPS</button>
                                    <button onClick={() => setMapViewMode('colonias')} className={`flex-1 sm:flex-none justify-center px-3 py-1.5 rounded-lg text-sm font-bold transition-colors ${mapViewMode === 'colonias' ? 'bg-cyan-600 text-white' : 'bg-slate-800 text-slate-400 hover:bg-slate-700'}`}>Tabla Sectores</button>
                                </div>
                            </div>

                            {mapViewMode === 'puntos' ? (
                                <div ref={mapContainerRef} className="flex-1 w-full bg-slate-950 rounded-xl border border-slate-800 relative z-0">
                                    <div className="absolute top-4 left-4 z-[400] pointer-events-none drop-shadow-lg hidden sm:block">
                                        <h3 className="text-white font-black text-lg drop-shadow-md bg-slate-900/60 px-3 py-1 rounded-lg backdrop-blur-sm border border-slate-700">Riesgo Sectorial SE {statsGlobales.semanaEpidemiologica}</h3>
                                    </div>
                                    <div className="absolute bottom-4 right-4 flex flex-col gap-2 bg-slate-900/90 p-3 rounded-lg border border-slate-700 backdrop-blur-md z-[400] shadow-xl">
                                        <span className="text-[10px] text-slate-400 font-bold mb-1 border-b border-slate-700 pb-1">RIESGO (PHM)</span>
                                        <span className="flex items-center gap-2 text-xs font-medium text-white"><div className="w-3 h-3 rounded-full bg-rose-500"></div> Crítico (&gt; 50)</span>
                                        <span className="flex items-center gap-2 text-xs font-medium text-white"><div className="w-3 h-3 rounded-full bg-orange-400"></div> Alto (&gt; 20)</span>
                                        <span className="flex items-center gap-2 text-xs font-medium text-white"><div className="w-3 h-3 rounded-full bg-cyan-600"></div> Bajo (&lt; 20)</span>
                                    </div>
                                    <MapaInteractivo data={mapDataWithSectorRisk} onMarkerClick={setSelectedHotspot} />
                                </div>
                            ) : (
                                <div className="h-full overflow-y-auto bg-slate-900/50 rounded-xl border border-slate-800">
                                    <table className="w-full text-left border-collapse relative z-10">
                                        <thead>
                                            <tr className="border-b border-slate-700 text-slate-400 text-xs sm:text-sm sticky top-0 bg-slate-900 shadow-md">
                                                <th className="p-3 font-semibold">Sector / Colonia</th>
                                                <th className="p-3 font-semibold text-center hidden sm:table-cell">Ovitrampas Revisadas</th>
                                                <th className="p-3 font-semibold text-center text-rose-400">Positivas</th>
                                                <th className="p-3 font-semibold text-center text-orange-400 hidden sm:table-cell">Huevos</th>
                                                <th className="p-3 font-semibold text-center text-amber-400 bg-amber-900/20">PHM Sector</th>
                                                <th className="p-3 font-semibold text-center">Nivel</th>
                                            </tr>
                                        </thead>
                                        <tbody className="divide-y divide-slate-800/50 text-xs sm:text-sm">
                                            {datosPorSector.map((sector, idx) => (
                                                <tr key={idx} className="hover:bg-slate-800/30 transition-colors">
                                                    <td className="p-3 font-medium text-white">{sector.nombre}</td>
                                                    <td className="p-3 text-center text-slate-300 hidden sm:table-cell">{sector.revisadas}</td>
                                                    <td className="p-3 text-center font-bold text-rose-400">{sector.positivas}</td>
                                                    <td className="p-3 text-center font-bold text-orange-400 hidden sm:table-cell">{sector.huevos}</td>
                                                    <td className="p-3 text-center font-bold text-amber-400 bg-amber-900/10">{sector.phm}</td>
                                                    <td className="p-3 text-center">
                                                        <span className={`px-2 py-1 rounded text-[10px] sm:text-xs font-bold uppercase ${sector.riesgoNivel === 'Crítico' ? 'bg-rose-500/20 text-rose-400' : sector.riesgoNivel === 'Alto' ? 'bg-orange-500/20 text-orange-400' : 'bg-cyan-500/20 text-cyan-400'}`}>{sector.riesgoNivel}</span>
                                                    </td>
                                                </tr>
                                            ))}
                                        </tbody>
                                    </table>
                                </div>
                            )}
                        </div>

                        <div className="w-full xl:w-96 flex flex-col gap-4 shrink-0">
                        {selectedHotspot && mapViewMode === 'puntos' ? (
                            <div className="bg-slate-900 p-6 rounded-2xl border border-cyan-500/30 shadow-[0_0_30px_rgba(6,182,212,0.1)] animate-fade-in">
                            <div className="flex items-start justify-between mb-4">
                                <h3 className="text-xl font-bold text-white break-words">{selectedHotspot.id}</h3>
                                <span className={`px-3 py-1 rounded-full text-xs font-black uppercase shrink-0 ${selectedHotspot.sectorRiesgo === 'Crítico' ? 'bg-rose-500/20 text-rose-400 border border-rose-500/50' : selectedHotspot.sectorRiesgo === 'Alto' ? 'bg-orange-500/20 text-orange-400 border border-orange-500/50' : 'bg-cyan-500/20 text-cyan-400 border border-cyan-500/50'}`}>
                                Riesgo: {selectedHotspot.sectorRiesgo}
                                </span>
                            </div>
                            <p className="text-slate-400 text-sm mb-6 flex items-center gap-2"><i className="fa-solid fa-location-dot"></i> S:{selectedHotspot.sector} | M:{selectedHotspot.manzana}</p>
                            <div className="grid grid-cols-2 gap-4 mb-6">
                                <StatBox label="Huevos Trampa" value={selectedHotspot.huevos} />
                                <StatBox label="PHM Sectorial" value={selectedHotspot.sectorPhm} isAlert={selectedHotspot.sectorRiesgo === 'Crítico'} />
                            </div>
                            <div className="border-t border-slate-800 pt-4">
                                <h4 className="text-sm text-slate-400 mb-3 flex items-center gap-2"><i className="fa-solid fa-bolt text-amber-400 text-[16px]"></i> Estrategia Recomendada</h4>
                                {selectedHotspot.sectorRiesgo === 'Crítico' ? (
                                <div className="space-y-2">
                                    <ActionTag text="INTERVENCIÓN INTEGRAL requerida." type="danger" />
                                    <ActionTag text="Barrido 100% + Nebulización Espacial." type="danger" />
                                </div>
                                ) : selectedHotspot.sectorRiesgo === 'Alto' ? (
                                <div className="space-y-2">
                                    <ActionTag text="OPERATIVO DE BARRIDO requerido." type="warning" />
                                    <ActionTag text="Trabajar 100% de manzanas del sector." type="warning" />
                                </div>
                                ) : (
                                <div className="space-y-2">
                                    <ActionTag text="TRATAMIENTO FOCALIZADO recomendado." type="info" />
                                    <ActionTag text="Visitar exclusivamente manzanas (+)." type="info" />
                                </div>
                                )}
                            </div>
                            </div>
                        ) : (
                            <div className="bg-slate-900/50 p-6 rounded-2xl border border-slate-800 h-[380px] flex flex-col items-center justify-center text-center text-slate-500 border-dashed">
                            <i className="fa-solid fa-map-location-dot text-[48px] mb-4 opacity-30"></i>
                            <p className="px-4">{mapViewMode === 'puntos' ? 'Selecciona una ovitrampa en el mapa satelital para ver la táctica recomendada.' : 'La tabla muestra la densidad de huevos consolidada por Sector.'}</p>
                            </div>
                        )}
                        </div>
                        </div>
                    )}
                    </div>
                </main>
                </div>
            );
        }

        function NavButton({ icon, label, active, onClick, disabled }) {
            return (
                <button onClick={disabled ? null : onClick} className={`flex items-center gap-2 px-4 py-2.5 rounded-xl transition-all duration-300 relative shrink-0 ${disabled ? 'opacity-30 cursor-not-allowed text-slate-600' : active ? 'text-cyan-400 bg-slate-800 border border-slate-700 shadow-md' : 'text-slate-400 hover:text-cyan-300 hover:bg-slate-800/50 border border-transparent'}`}>
                    <div className={`${active ? 'drop-shadow-[0_0_8px_rgba(34,211,238,0.8)]' : ''}`}>{icon}</div>

                </button>
            );
        }

        function StatBox({ label, value, isAlert }) {
            return (
                <div className={`p-3 rounded-xl border ${isAlert ? 'bg-rose-500/10 border-rose-500/30' : 'bg-slate-800/50 border-slate-700'}`}>
                <p className="text-xs text-slate-400 mb-1 font-medium uppercase tracking-wider">{label}</p><p className={`text-xl font-black truncate ${isAlert ? 'text-rose-400' : 'text-white'}`}>{value}</p>
                </div>
            );
        }

        function IndicatorCard({ title, name, value, formula, explanation, colorTheme }) {
            const themes = { rose: 'bg-rose-500/10 border-rose-500/30 text-rose-400 from-rose-900/40', orange: 'bg-orange-500/10 border-orange-500/30 text-orange-400 from-orange-900/40', amber: 'bg-amber-500/10 border-amber-500/30 text-amber-400 from-amber-900/40', emerald: 'bg-emerald-500/10 border-emerald-500/30 text-emerald-400 from-emerald-900/40', cyan: 'bg-cyan-500/10 border-cyan-500/30 text-cyan-400 from-cyan-900/40' };
            const themeClass = themes[colorTheme] || themes.cyan;
            return (
                <div className={`bg-slate-900 border rounded-2xl p-6 relative overflow-hidden group hover:bg-gradient-to-br ${themeClass} transition-all duration-300`}>
                <div className="flex justify-between items-start mb-4">
                    <div><h3 className="text-2xl font-black text-white">{title}</h3><p className="text-xs font-bold uppercase tracking-wider opacity-70 mt-1">{name}</p></div>
                    <div className="text-3xl font-black drop-shadow-md">{value}</div>
                </div>
                <div className="mb-4 bg-slate-950/50 p-3 rounded-lg border border-slate-800 flex items-center gap-3">
                    <i className="fa-solid fa-calculator opacity-50 shrink-0 text-[18px]"></i><span className="text-xs font-mono text-slate-300 break-words">{formula}</span>
                </div>
                <div className="flex items-start gap-2 text-sm text-slate-400"><i className="fa-solid fa-circle-info shrink-0 mt-0.5 text-[16px]"></i><p className="leading-relaxed">{explanation}</p></div>
                </div>
            );
        }

        function ActionTag({ text, type }) {
            const styles = { danger: 'bg-rose-950/50 border-rose-800 text-rose-300', warning: 'bg-orange-950/50 border-orange-800 text-orange-300', info: 'bg-cyan-950/50 border-cyan-800 text-cyan-300', primary: 'bg-indigo-950/50 border-indigo-800 text-indigo-300' };
            return (
                <div className={`flex items-start gap-2 p-2 border rounded-lg text-xs font-medium leading-tight ${styles[type]}`}>
                <i className="fa-solid fa-chevron-right shrink-0 mt-0.5 text-[12px]"></i> {text}
                </div>
            );
        }

        function TacticalCard({ title, priority, target, icon, colorTheme, action, onClick }) {
            const themes = { 
                rose: { bgIcon: 'bg-rose-500/20 text-rose-400', badge: 'bg-rose-500/20 text-rose-400 border-rose-500/30', glow: 'bg-rose-500/10', btnHover: 'hover:bg-rose-600 hover:border-rose-500' }, 
                orange: { bgIcon: 'bg-orange-500/20 text-orange-400', badge: 'bg-orange-500/20 text-orange-400 border-orange-500/30', glow: 'bg-orange-500/10', btnHover: 'hover:bg-orange-600 hover:border-orange-500' }, 
                cyan: { bgIcon: 'bg-cyan-500/20 text-cyan-400', badge: 'bg-cyan-500/20 text-cyan-400 border-cyan-500/30', glow: 'bg-cyan-500/10', btnHover: 'hover:bg-cyan-600 hover:border-cyan-500' }, 
                indigo: { bgIcon: 'bg-indigo-500/20 text-indigo-400', badge: 'bg-indigo-500/20 text-indigo-400 border-indigo-500/30', glow: 'bg-indigo-500/10', btnHover: 'hover:bg-indigo-600 hover:border-indigo-500' } 
            };
            const theme = themes[colorTheme] || themes.cyan;
            return (
                <div className="bg-slate-900 border border-slate-800 rounded-2xl p-6 relative overflow-hidden group hover:border-slate-700 transition-all">
                <div className={`absolute -top-10 -right-10 w-32 h-32 blur-3xl rounded-full transition-all ${theme.glow}`}></div>
                <div className="flex justify-between items-start mb-6 relative z-10"><div className={`p-3 rounded-xl border border-transparent ${theme.bgIcon}`}>{icon}</div><span className={`px-2 py-1 rounded border text-xs font-black uppercase tracking-wider ${theme.badge}`}>{priority}</span></div>
                <div className="relative z-10"><h3 className="text-lg font-bold text-white mb-2">{title}</h3><p className="text-sm text-slate-400 mb-6 flex items-center gap-2 font-medium"><i className="fa-solid fa-crosshairs text-[16px]"></i> {target}</p><button onClick={onClick} className={`w-full py-3 rounded-xl text-white font-bold transition-all shadow-lg text-sm bg-slate-800 border border-slate-700 ${theme.btnHover}`}>{action}</button></div>

            );
